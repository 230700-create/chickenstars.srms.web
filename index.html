// WEB RTC CONFIG const config = { iceServers: [{ urls: "stun:stun.l.google.com:19302" }] }; let pc = new RTCPeerConnection(config); let dc; // 1. Initialize Camera async function initCamera() { try { const stream = await navigator.mediaDevices.getUserMedia({ video: true, audio: true }); document.getElementById('local-v').srcObject = stream; stream.getTracks().forEach(track => pc.addTrack(track, stream)); // Add tracks to peer connection } catch (e) { appendMsg("System", "Camera/Mic access denied.", "system"); } } initCamera(); // Handle incoming Remote Stream pc.ontrack = e => { if (e.streams && e.streams[0]) { document.getElementById('remote-v').srcObject = e.streams[0]; } }; // Data Channel Handlers for Chat function setupDataHandlers() { dc.onopen = () => { document.body.classList.add('active'); // CSS hook for status document.getElementById('status-text').innerText = "Link Established"; appendMsg("System", "Secure Channel Open.", "system"); }; dc.onmessage = e => appendMsg("Peer", e.data, "peer"); } // Handle incoming Data Channel (for the receiver/Peer B) pc.ondatachannel = e => { dc = e.channel; setupDataHandlers(); }; // STEP 1: Create Offer (Peer A) async function createOffer() { dc = pc.createDataChannel("chat"); // Peer A creates the channel setupDataHandlers(); const offer = await pc.createOffer(); await pc.setLocalDescription(offer); document.getElementById('status-text').innerText = "Gathering Candidates..."; // Wait for ICE candidates to bundle into the SDP pc.onicecandidate = e => { if (!e.candidate) { // All candidates gathered document.getElementById('my-signal').value = JSON.stringify(pc.localDescription); document.getElementById('status-text').innerText = "Signal Ready"; } }; } // STEP 2: Handle Friend's Signal (Connect) async function connectFriend() { const input = document.getElementById('friend-signal').value.trim(); if (!input) return alert("Please paste a signal code first."); try { const signal = JSON.parse(input); if (signal.type === "offer") { // Receiver Logic (Peer B) await pc.setRemoteDescription(new RTCSessionDescription(signal)); const answer = await pc.createAnswer(); await pc.setLocalDescription(answer); pc.onicecandidate = e => { if (!e.candidate) { document.getElementById('my-signal').value = JSON.stringify(pc.localDescription); document.getElementById('status-text').innerText = "Answer Generated"; } }; } else if (signal.type === "answer") { // Sender Logic (Peer A) await pc.setRemoteDescription(new RTCSessionDescription(signal)); } } catch (err) { alert("Invalid Signal Code."); } } // Messaging Logic function sendMsg() { const input = document.getElementById('msg-in'); if (dc && dc.readyState === "open" && input.value) { dc.send(input.value); appendMsg("You", input.value, "me"); input.value = ""; } } function appendMsg(user, text, type) { const box = document.getElementById('chat-box'); const msgDiv = document.createElement('div'); msgDiv.className = `p-2 rounded-lg text-xs ${type === 'me' ? 'bg-cyan-900/30 ml-8 text-right' : 'bg-white/5 mr-8'}`; msgDiv.innerHTML = `${user}${text}`; box.appendChild(msgDiv); box.scrollTop = box.scrollHeight; } function copySignal() { const el = document.getElementById('my-signal'); el.select(); document.execCommand('copy'); }
